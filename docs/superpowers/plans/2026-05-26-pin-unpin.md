```swift
    /// Sets the `pinned` flag on the item with the given id.
    /// Pinned items are excluded from all retention purges.
    func setPinned(itemId: Int64, pinned: Bool) throws {
        _ = try queue.write { db in
            try Item.filter(Item.Columns.id == itemId)
                .updateAll(db, [Item.Columns.pinned.set(to: pinned)])
        }
        postChange()
    }
```

```swift
    private func handlePin(item: Item, pinned: Bool) {
        guard let id = item.id else { return }
        do { try store.setPinned(itemId: id, pinned: pinned) }
        catch { Log.drawer.error("pin failed: \(error.localizedDescription, privacy: .public)") }
    }
```
